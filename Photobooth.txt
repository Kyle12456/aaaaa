<script>

const video = document.getElementById("video");
const strip = document.getElementById("strip");
// countdown + flash overlays (created in DOM below)
const countdownEl = document.getElementById('countdown');
const flashEl = document.getElementById('flash');

navigator.mediaDevices.getUserMedia({ video: true })
	.then(stream => {
		video.srcObject = stream;
		// some browsers require an explicit play call
		video.play().catch(() => {});
	})
	.catch(err => {
		console.error('Camera error:', err);
		document.getElementById('message').innerText = 'Camera unavailable or permission denied.';
	});

function takePhoto(){

	// ensure video has data
	if (video.readyState < 2) {
		alert('Camera not ready ‚Äî try again');
		return;
	}

	// use an offscreen canvas so we don't need a DOM canvas element
	const canvas = document.createElement('canvas');
	const context = canvas.getContext('2d');

	canvas.width = 240;
	canvas.height = 180;

	context.drawImage(video, 0, 0, 240, 180);

	/* add to photostrip: wrap in figure with caption */
	let img = document.createElement('img');
	img.src = canvas.toDataURL('image/png');

	const figure = document.createElement('figure');
	const caption = document.createElement('figcaption');
	caption.innerText = new Date().toLocaleTimeString();
	figure.appendChild(img);
	figure.appendChild(caption);
	strip.appendChild(figure);

}

// Trigger capture with optional countdown
function triggerCapture(){
	const secs = parseInt(document.getElementById('timerSelect').value, 10) || 0;
	if(secs > 0){
		startCountdown(secs);
	} else {
		takePhoto();
	}
}

function startCountdown(seconds){
	if(!countdownEl) return takePhoto();
	let remaining = seconds;
	countdownEl.style.display = 'block';
	countdownEl.innerText = remaining;
	const interval = setInterval(()=>{
		remaining -= 1;
		if(remaining > 0){
			countdownEl.innerText = remaining;
		} else {
			clearInterval(interval);
			countdownEl.style.display = 'none';
			// flash effect
			if(flashEl){
				flashEl.style.transition = 'none';
				flashEl.style.opacity = '0.95';
				setTimeout(()=>{ flashEl.style.transition = 'opacity 360ms ease'; flashEl.style.opacity = '0'; }, 60);
			}
			takePhoto();
		}
	}, 1000);
}

function saveMemory(){

const letter = document.getElementById("letter").value;
const date = document.getElementById("futureDate").value;

if(letter === "" || date === ""){
alert("Complete everything first!");
return;
}

const memory = {
letter: letter,
date: date,
photos: strip.innerHTML
};

localStorage.setItem("futureMemory", JSON.stringify(memory));

document.getElementById("message").innerText =
"üíø Memory sealed for future you.";
}
